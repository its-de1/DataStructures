class Solution:
    def isPalindrome(self, head):
        #code here
            fast,slow=head,head
            rev=None
            while fast and fast.next:
                fast=fast.next.next
                newslow=slow.next
                slow.next=rev
                rev=slow
                slow=newslow
            if fast:
                slow=slow.next
            while slow:
                if slow.data!=rev.data:
                    return False
                slow=slow.next
                rev=rev.next
            return True
